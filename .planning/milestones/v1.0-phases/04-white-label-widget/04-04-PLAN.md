**Step 3: Add error handling for auth failures.**

In `api/client.ts`, add auth failure handling:

```typescript
export async function apiClient(path: string, apiKey: string, options: RequestInit = {}): Promise<Response> {
  const res = await fetch(`${API_BASE}/api/v1${path}`, {
    ...options,
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${apiKey}`,
      ...(options.headers || {}),
    },
  });

  // Surface auth errors clearly
  if (res.status === 401) {
    throw new Error('Invalid API key — check your data-api-key attribute');
  }
  if (res.status === 403) {
    throw new Error('API key not verified — please verify your email first');
  }

  return res;
}
```

**Step 3: Add analytics retrieval endpoint.**

Add `GET /widget/analytics` — allows the contractor to see their widget analytics. Requires API key or JWT auth.

```python
@widget_router.get("/analytics")
@limiter.limit("10/minute")
async def get_analytics(
    request: Request,
    tenant: Tenant = Depends(get_current_tenant),
    days: int = 30,
):
    """Get widget analytics for the last N days (default 30)."""
    from datetime import datetime, timedelta, timezone
    start_date = (datetime.now(timezone.utc) - timedelta(days=days)).strftime("%Y-%m-%d")
    db = get_database()
    cursor = db["widget_analytics"].find(
        {"tenant_id": tenant.tenant_id, "date": {"$gte": start_date}},
        {"_id": 0},
    ).sort("date", -1)
    results = await cursor.to_list(length=days)
    return {"analytics": results, "days": days}
```
